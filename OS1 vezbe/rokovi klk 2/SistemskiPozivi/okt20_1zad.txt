int multispawn(int number, const char* path, const char* args[]){
    int ret = 0;
    const char* childArgs[3];
    childArgs[0] = path;
    childArgs[2] = NULL;
    for(int i=0; i<number; i++){
        char* childArgs[1] = args[i];
        int status = spawnvp(P_NOWAIT, path, argv)
        if(status >= 0){
            ret++;
        }
    }
    return ret;
}
