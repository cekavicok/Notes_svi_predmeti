typedef unsighed long Time;
const Time MAXTIME = (Time) - 1;


class Resource {
public:
    Resource ();
    int acquire();
    void release();
    void block();
    void deblock();
private:
    Thread* holder;
    Time last;
    ThreadQueue blocked;
};


int Resource::acquire(){
    lock();
    if(!holder){
        holder = Thread::running;
        last = holder->timestamp;
        unlock();
        return 1;
    }
    else{
        if(Thread::running->timestamp >= last){
            unlock();
            return 0;
        }
        else{
            last = Thread::running->timestamp;
            block();
            unlock();
            return 1;
        }
    }
}

void Resource::release(){
    lock();
    Thread* thr = blocked.get();
    if(thr){
        holder = thr;
        deblock();
    } else{
        holder = 0;
        last = MAXTIME;
    }
    unlock();
}
