class FreeFrames {
public:
    FreeFrames(void* mem, size_t size);
    void* alloc();
    void free(void* frame);
private:
    char* head;
};

void FreeFrames::FreeFrames(void* mem, size_t size){
    char* frame = head = (char*)mem;

    // gradi ulancanu listu: svaki okvir na pocetku cuva adresu sledeceg
    // frame = ... = frame+PAGE_SIZE: istovremeno upisuje adresu sledeceg i pomera frame
    //frame + PAGE_SIZE = 0x04  adresa sledeceg okvira
    //*(char**)frame = 0x04     na lokaciju 0x00 upiši 0x04
    //frame = 0x04              frame se pomera na sledeci
    // Stanje: [0x00 → 0x04] [0x04] [0x08]

    for(int i=0; i<size-1; i++){
        frame = (*(char**)frame) = frame + PAGE_SIZE;
    }
    *(char**)frame = 0; // poslednji okvir pokazuje na NULL
}

void* FreeFrames::alloc(){
    void* ret = head;
    if(head) {
        head = *(char**)head;
    }
    return;
}


void free(void* frame){
    *(char**)frame = head;
    head = (char*)frame;
}
